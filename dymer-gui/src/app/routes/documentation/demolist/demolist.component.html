The following is an example of how to setup a simple Dymer Viewer.

Create a js function named mainDymerView(), define the json that configures the Viewer and pass it in the funtion drawEntities()

<script>
    var jsonDymerConfig = {{ '{' }}
        query: {{ '{' }} // define the query to do in dymer
            "instance": {{ '{' }}
                "index": index-entity,
                "type":  type-entity
            }
        },
        endpoint: 'entity.search', // set the endpoint to execute the query of entities
        viewtype: 'teaserlist', // set to use the 'Preview in List' template
        target: {{ '{' }}
            teaserlist: {{ '{' }} // configure where and how to render the preview of entities
                    id: "#cont-MyList", // the entities will be render inside the element with id "#cont-MyList"
                    action: "html", // set the method to insert the content (html/append/prepend)
                    reload: false // if false the query will be executed only on page load
            },
            fullcontent: {{ '{' }}// configure where and how to render the detail of an entity
                id: "#cont-MyList", // the entity will be rendered inside the element with id "#cont-MyList"
                action: "html" // set the method to insert the content (html/append/prepend)
            }
        }
    };
    function mainDymerView() {{ '{' }}
        drawEntities(jsonDymerConfig);
    }
 </script>

Import the dymer.viewer.js with id 'dymerurl' and set the <dymerip> 

 <script id="dymerurl" src="<dymerip>/public/cdn/js/dymer.viewer.js"> </script>

Create a DOM element where render the entities and another which onclick invokes the function drawEntities() to redraw them

 <span class="btn btn-mini btn-info " onclick="drawEntities(jsonDymerConfig)"> My List </span>
 <div id="cont-MyList"></div>

Dymer Viewer imports Bootstrap v4 and jQuery 3, if you use previous versions add a json object called dymerconf as below to exclude the import. To set the 'user id' and 'group id' add the variables d_uid and d_gid. 

 var d_uid = '<user-id>'; // user id
 var d_gid = '<group-id>'; // group id
 var dymerconf = {{ '{' }} // disable import bootstrap & jquery
    notImport : ["bootstrap","jquery"]
 };

Create a DOM element where render the entities and more elements which on onclick event invokes the function switchQuery() to redraw them with different query

 var dymerQueries = [
   {{ '{' }}// 1* query
        "match": {{ '{' }}
            "_index": "geopoint"
        }
    },
    {{ '{' }}// 2* query
        "match": {{ '{' }}
            "_index": "multir"
        }
    },
    {{ '{' }}// 3* query
        "terms": {{ '{' }}
            "_index": ["multir", "geopoint" ]
        }
    }
    ];

<span class="btn btn-mini btn-info " onclick="switchQuery(dymerQueries[0])"> My List 1 </span>
<div id="cont-MyList"></div>
<span class="btn btn-mini btn-info " onclick="switchQuery(dymerQueries[1])"> My List 2</span>
<div id="cont-MyList"></div>
<span class="btn btn-mini btn-info " onclick="switchQuery(dymerQueries[2])"> My List 3</span>
<div id="cont-MyList"></div>
